# definition for profile
@auth.requires_login()
def profile():
    p = db.auth_user(request.args(0))
    fname = p.first_name
    lname = p.last_name
    email = p.email
    ptext = p.profiletext
    pmage = p.profileimage
    return dict(fname=fname, lname=lname, email=email, ptext=ptext, pmage=pmage)
